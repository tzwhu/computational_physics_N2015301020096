import random
import math
import pylab as pl
from mpl_toolkits.mplot3d import Axes3D

class three_random_walk:
    def __init__(self, step_numbers = 100):
        self.N = step_numbers
        self.x = [0]
        self.y = [0]
        self.z = [0]
        
    def simulation(self):
        for i in range(self.N):
            rnd = random.random()
            if 0<rnd<1/6:
                self.x.append(self.x[i]+1)
                self.y.append(self.y[i])
                self.z.append(self.z[i])
            elif 1/6<rnd<1/3:
                self.x.append(self.x[i]-1)
                self.y.append(self.y[i])
                self.z.append(self.z[i])
            elif 1/3<rnd<1/2:
                self.x.append(self.x[i])
                self.y.append(self.y[i]+1)
                self.z.append(self.z[i])
            elif 1/2<rnd<2/3:
                self.x.append(self.x[i])
                self.y.append(self.y[i]-1)
                self.z.append(self.z[i])
            elif 2/3<rnd<5/6:
                self.x.append(self.x[i])
                self.y.append(self.y[i])
                self.z.append(self.z[i]+1)
            else:
                self.x.append(self.x[i])
                self.y.append(self.y[i])
                self.z.append(self.z[i]-1)

    def plot(self):
        fig = pl.figure(figsize=(10,10))
        ax1 = fig.add_subplot(111, projection='3d')
        ax1.scatter(self.x,self.y,self.z,c='darkred',marker='o')
#        pl.plot(self.x, self.y,'ok')
#        pl.plot(self.x, self.y,'c')
#        坐标轴
        ax1.set_zlabel('$z$',size=20) 
        ax1.set_ylabel('$y$',size=20)
        ax1.set_xlabel('$x$',size=20)
        ax1.set_title('Random walk in three dimensions',size=26)
